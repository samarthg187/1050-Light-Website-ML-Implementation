Project Overview

Hardware components, sensor firmware, and the CSV data format produced by the device.

Hardware

Sensor Components

ComponentPartRole
Spectral SensorAS7341Captures 8 visible light bands + NIR + Clear at 2-second intervals via I²C. Gain set to 16× for urban light levels (saturation ~1600 lux).
Real-Time ClockDS1302Timestamps each reading with date and time via 3-wire interface
Storage SD CardWrites readings to a CSV file named by boot date (e.g. 0315.csv)
MicrocontrollerESP32Coordinates all components; pins SDA=33, SCL=32, CS=5

Data Format

CSV Output

Each session produces one CSV file. The sensor writes one row every 2 seconds with a header on first boot:

Date,Time,F1,F2,F3,F4,F5,F6,F7,F8,NIR,Clear,Lux_Est

FieldFormatExampleDescription
Date yyyy/mm/dd2024/03/15Calendar date with slashes
Time hh:mm:ss 07:30:00 24-hour time with colons
F1–F8 integer 120 Visible spectral band intensities (415 nm violet → 680 nm red)
NIR integer 88 Near-infrared channel intensity
Clear integer 450 Broadband clear channel — used for map colour-coding
Lux_Est float 17.632 Estimated illuminance in lux, derived from the Clear channel using the formula: Clear / (GAIN × TINT) × LUX_FACTOR

Firmware

Sensor Code

Arduino sketch running on the ESP32 (firmware v2). Initialises the RTC, AS7341, and SD card on boot, then logs one row every 2 seconds.

light_monitor_v2.inoC++
/*
 * Light Pollution Monitor — Firmware v2
 * ======================================
 * Hardware: ESP32 WROOM + Adafruit AS7341 + DS1302 RTC + MicroSD
 *
 * GAIN CHANGE from original firmware
 * -----------------------------------
 * The original code used AS7341_GAIN_256X.  That gain was designed for
 * dark-sky astronomy — it saturates (maxes out) at roughly 100 lux, which
 * is less bright than a streetlight seen from 5 metres away.
 *
 * This project measures point-source light pollution in an urban environment:
 *   • Glare from streetlights at 2–5 m         → ~100–400 lux
 *   • Light trespass at a property boundary     → 0.5–20 lux
 *   • Skyglow (diffuse, sensor tilted upward)   → 0.05–5 lux
 *
 * AS7341_GAIN_16X gives:
 *   Saturation at: ~1600 lux  (handles all practical urban scenes)
 *   Min detectable: ~0.25 lux (still resolves trespass below the 2-lux limit)
 */

#include <Wire.h>
#include <ThreeWire.h>
#include <RtcDS1302.h>
#include <Adafruit_AS7341.h>
#include <SPI.h>
#include <SD.h>

// ── Pin assignments ───────────────────────────────────────────────────────────
const int SDA_PIN    = 33;
const int SCL_PIN    = 32;
const int RTC_IO     = 26;
const int RTC_SCLK   = 25;
const int RTC_RST    = 27;
const int chipSelect =  5;

// ── Lux calibration ──────────────────────────────────────────────────────────
// lux ≈ Clear_ADC / (GAIN × TINT_S) × LUX_FACTOR
// GAIN_MULT must match the gain constant set below (16 for AS7341_GAIN_16X).
// LUX_FACTOR: validate against a reference meter. 0.110 is a starting point.
const float GAIN_MULT  =   16.0f;
const float TINT_S     =    0.2808f;
const float LUX_FACTOR =    0.110f;

// ── Saturation threshold ──────────────────────────────────────────────────────
const uint16_t SAT_WARN = 65400;

ThreeWire            myWire(RTC_IO, RTC_SCLK, RTC_RST);
RtcDS1302<ThreeWire> Rtc(myWire);
Adafruit_AS7341      as7341;

char sessionFilename[16];
bool needHeader = false;

float estimateLux(uint16_t clearCounts) {
    return (float)clearCounts / (GAIN_MULT * TINT_S) * LUX_FACTOR;
}

void setup() {
    Serial.begin(115200);

    Rtc.Begin();
    Rtc.SetIsWriteProtected(false);
    Rtc.SetIsRunning(true);

    RtcDateTime bootTime = Rtc.GetDateTime();
    sprintf(sessionFilename, "/%02u%02u.csv", bootTime.Month(), bootTime.Day());

    Wire.begin(SDA_PIN, SCL_PIN);
    if (as7341.begin()) {
        as7341.setATIME(100);
        as7341.setASTEP(999);
        as7341.setGain(AS7341_GAIN_16X);  // 16× — saturation ~1600 lux
    }

    if (SD.begin(chipSelect)) {
        needHeader = !SD.exists(sessionFilename);
    }
}

void loop() {
    RtcDateTime now = Rtc.GetDateTime();
    uint16_t readings[12];

    if (as7341.readAllChannels(readings)) {
        char ts[25];
        sprintf(ts, "%04u/%02u/%02u,%02u:%02u:%02u",
                now.Year(), now.Month(), now.Day(),
                now.Hour(), now.Minute(), now.Second());

        float lux = estimateLux(readings[9]);  // Clear = index 9

        // Saturation warning (serial only — data still logged)
        bool saturated = false;
        for (int i = 0; i < 10; i++) {
            if (readings[i] >= SAT_WARN) { saturated = true; break; }
        }
        if (saturated) Serial.printf("[!] SAT — move 2–5 m from source (lux~%.0f)\n", lux);

        // Write header on first boot
        if (needHeader) {
            File f = SD.open(sessionFilename, FILE_WRITE);
            if (f) {
                f.println("Date,Time,F1,F2,F3,F4,F5,F6,F7,F8,NIR,Clear,Lux_Est");
                f.close();
                needHeader = false;
            }
        }

        // Append data row
        File dataFile = SD.open(sessionFilename, FILE_APPEND);
        if (dataFile) {
            dataFile.print(ts);
            for (int i = 0; i < 10; i++) {
                dataFile.print(","); dataFile.print(readings[i]);
            }
            dataFile.printf(",%.3f\n", lux);
            dataFile.close();
        }
    }

    delay(2000);  // 2-second logging interval
}
